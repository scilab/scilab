function lnk=scicos_link(v1,v2,v3,v4,v5,v6,v7,v8)

  if exists('xx','local')==0 then xx=[],end
  if exists('yy','local')==0 then yy=[],end
  if exists('id','local')==0 then id='drawlink',end
  if exists('thick','local')==0 then thick=[0,0],end
